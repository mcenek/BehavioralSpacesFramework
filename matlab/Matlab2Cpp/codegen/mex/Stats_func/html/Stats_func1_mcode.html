1function [ rmat ] = Stats_func( HistFutTable,threshold )
 2
 3
 4emptyrows = all (~HistFutTable,2);
 5emptyrows = find (emptyrows == 1);
 6
 7
 8chi2L = 1:size(HistFutTable);
 9sums = sum(HistFutTable,2);
10firstleader = find (sums == max(sums));
11firstleader = firstleader (1);
12%The first leader is seeded in order to ensure that it is not an extremely
13%sparse row of the matrix - which creates problems with the chi square test
14%of independence
15
16% %Statistical Correlation
17for its3 = 1:size(emptyrows,1)
18    x = emptyrows (its3);
19    chi2L = chi2L(chi2L ~= x);
20end
21rmat = zeros(1,size(HistFutTable,1))';
22%leaders = chi2L (randmat);
23leaders = firstleader;
24rmat (firstleader) = -1;
25leadercomp = zeros (1,size(HistFutTable,1));
26chi2L = chi2L (chi2L ~= firstleader);
27chis = zeros (2,size(HistFutTable,2));
28% p2chi = zeros (1,size(RData,2));
29while size(chi2L) > 0
30    randmat = ceil(size(chi2L,2) * rand(1));
31    randmat = chi2L(randmat);
32    for its = leaders
33        chis = [HistFutTable(its,:);HistFutTable(randmat,:)];
34        leadercomp (its) = chisquared(chis);
35    end
36    [M,I] = max (leadercomp);
37    if M > threshold
38        rmat (randmat) = I;
39        chi2L = chi2L (chi2L ~= randmat);
40    else
41        rmat(randmat) = -1;
42        leaders = horzcat (leaders,randmat);   %#ok
43        chi2L = chi2L (chi2L ~= randmat);
44    end
45end
