1function [ y ] = comptable( range,dims )
 2%Generates a reference table for the size of the square transition table
 3%RData in the main program. Useful for finding which combinations of range
 4%and dimensions are computationally feasible, and for preallocating
 5%matrices which store this data. 
 6range = range+1;
 7res = zeros (range,dims);
 8res (1:range,1) = 1;
 9res (1,1:dims) = 1;
10
11for i = 2:range
12    for i2 = 2:dims
13        res (i,i2) = res(i-1,i2) + res (i,i2-1);
14    end
15end
16y = res;
17end
